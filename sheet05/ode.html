import numpy as np
import matplotlib.pyplot as plt

# Constants
delta_tau = 0.01
total_time = 10.0
num_steps = int(total_time / delta_tau)

# Initial conditions
s0 = np.array([1.0, 0.0])  # Initial position vector
w0 = np.array([0.0, 1.0])  # Initial velocity vector

s = np.zeros((num_steps, 2))
w = np.zeros((num_steps, 2))

s[0] = s0
w[0] = w0

# Forward Euler integration
for i in range(1, num_steps):
    # Update position
    s[i] = s[i-1] + delta_tau * w[i-1]

    # Update velocity
    s_magnitude = np.linalg.norm(s[i-1])
    w[i] = w[i-1] - delta_tau * s[i-1] / (s_magnitude ** 3)

# Plot the result
plt.figure(figsize=(8, 8))
plt.plot(s[:, 0], s[:, 1], label='Orbit')
plt.scatter([0], [0], color='red', label='Center of Mass')
plt.xlabel('x')
plt.ylabel('y')
plt.title('Circular Orbit of Two Point Masses')
plt.legend()
plt.grid(True)
plt.axis('equal')
plt.show()

def compute_lrl_vector(s, w):
    L = np.cross(s, w)  # Specific angular momentum (scalar in 2D)
    A_x = w[1] * L - s[0] / np.linalg.norm(s)
    A_y = -w[0] * L - s[1] / np.linalg.norm(s)
    A = np.array([A_x, A_y])
    return A

def compute_eccentricity(s, w):
    A = compute_lrl_vector(s, w)
    e = np.linalg.norm(A)
    return e

def integrate_orbit_leap_frog(s0, w0, delta_tau, total_time):

    s = np.zeros((num_steps, 2))
    w = np.zeros((num_steps, 2))
    s[0] = s0
    w[0] = w0

    # Half-step velocity update
    s_magnitude = np.linalg.norm(s[0])
    w_half = w[0] - 0.5 * delta_tau * s[0] / (s_magnitude ** 3)

        s[i] = s[i-1] + delta_tau * w_half
        s_magnitude = np.linalg.norm(s[i])
        w[i] = w_half - delta_tau * s[i] / (s_magnitude ** 3)
        w_half = w[i] - delta_tau * s[i] / (s_magnitude ** 3)

    return s

3. The Simple Harmonic Oscillator

In[10]:
k = 1.0  # Spring constant
m = 1.0  # Mass
omega = np.sqrt(k / m)  # Oscillation frequency

y0 = 1.0  # Initial position
v0 = 0.0  # Initial velocity

h_values = [0.1, 0.01]
total_time = 10.0

def integrate_harmonic_oscillator(y0, v0, h, total_time, k, m):
    num_steps = int(total_time / h)
    y = np.zeros(num_steps)
    v = np.zeros(num_steps)
    t = np.linspace(0, total_time, num_steps)

    y[0] = y0
    v[0] = v0

    for n in range(1, num_steps):
        y[n] = y[n-1] + h * v[n-1]
        v[n] = v[n-1] - h * (k / m) * y[n-1]

    return t, y

plt.figure(figsize=(10, 6))
for h in h_values:
    t, y = integrate_harmonic_oscillator(y0, v0, h, total_time, k, m)
    plt.plot(t, y, label=f'h = {h}')

t_analytic = np.linspace(0, total_time, 1000)
y_analytic = np.cos(omega * t_analytic)
plt.plot(t_analytic, y_analytic, 'k--', label='Analytic Solution')

plt.xlabel('Time')
plt.ylabel('Position y(t)')
plt.title('Time Evolution of the Simple Harmonic Oscillator')
plt.legend()
plt.grid(True)
plt.show()

G = 1.0  # Gravitational constant
M = 1.0  # Mass of the bodies
mu = 1.0  # Reduced mass

def compute_energy(s, w):
    r = np.linalg.norm(s)
    K = 0.5 * mu * np.linalg.norm(w) ** 2
    U = -G * M * mu / r
    E = K + U
    return E

def compute_relative_energy_error(E, E0):
    return np.abs(E - E0) / E0

def integrate_orbit_forward_euler(s0, w0, delta_tau, num_steps):
    s = np.zeros((num_steps, 2))
    w = np.zeros((num_steps, 2))
    s[0] = s0
    w[0] = w0

    E0 = compute_energy(s0, w0)
    energy_errors = np.zeros(num_steps)

    for i in range(1, num_steps):
        s[i] = s[i-1] + delta_tau * w[i-1]
        s_magnitude = np.linalg.norm(s[i-1])
        w[i] = w[i-1] - delta_tau * s[i-1] / (s_magnitude ** 3)

        E = compute_energy(s[i], w[i])
        energy_errors[i] = compute_relative_energy_error(E, E0)

    return energy_errors

def integrate_orbit_leap_frog(s0, w0, delta_tau, num_steps):
    s = np.zeros((num_steps, 2))
    w = np.zeros((num_steps, 2))
    s[0] = s0
    w[0] = w0

    E0 = compute_energy(s0, w0)
    energy_errors = np.zeros(num_steps)

    # Half-step velocity update
    s_magnitude = np.linalg.norm(s[0])
    w_half = w[0] - 0.5 * delta_tau * s[0] / (s_magnitude ** 3)

    for i in range(1, num_steps):
        s[i] = s[i-1] + delta_tau * w_half
        s_magnitude = np.linalg.norm(s[i])
        w[i] = w_half - delta_tau * s[i] / (s_magnitude ** 3)
        w_half = w[i] - delta_tau * s[i] / (s_magnitude ** 3)

        E = compute_energy(s[i], w[i])
        energy_errors[i] = compute_relative_energy_error(E, E0)

    return energy_errors

s0 = np.array([1.0, 0.0])
w0_values = [0.5, 1.0, 1.5]

delta_tau_values = np.logspace(-4, -1, 10)

num_orbits = 2
total_time = num_orbits * 2 * np.pi
num_steps = int(total_time / delta_tau_values[0])

plt.figure(figsize=(10, 6))
for w0 in w0_values:
    w0_vec = np.array([0.0, w0])
    energy_errors_euler = np.zeros(len(delta_tau_values))
    energy_errors_leap_frog = np.zeros(len(delta_tau_values))

    for i, delta_tau in enumerate(delta_tau_values):
        num_steps = int(total_time / delta_tau)

        # Forward Euler
        energy_errors_euler[i] = np.max(integrate_orbit_forward_euler(s0, w0_vec, delta_tau, num_steps))

        # Leap-frog
        energy_errors_leap_frog[i] = np.max(integrate_orbit_leap_frog(s0, w0_vec, delta_tau, num_steps))

    plt.loglog(delta_tau_values, energy_errors_euler, '--', label=f'Forward Euler: w0 = {w0}')
    plt.loglog(delta_tau_values, energy_errors_leap_frog, '-', label=f'Leap-frog: w0 = {w0}')
